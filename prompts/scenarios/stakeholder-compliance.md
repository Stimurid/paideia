# Проверка на нормативку и этику

Где проект может столкнуться с ФГОС, ФЗ «Об образовании», обработкой ПДн,
этическими кодексами, академической честностью.

```
{
  "regulatory_risks": [
    {
      "norm_area": "ФГОС|ФЗ-273|152-ФЗ ПДн|этика|академ.честность|...",
      "specific_provision": "<конкретная статья/требование, если знаешь>",
      "risk": "<в чём конкретно риск>",
      "severity": 1-5,
      "mitigations": ["<что можно сделать>"],
      "corpus_precedents": ["<case_id с похожим столкновением>"]
    }
  ],
  "ethics_concerns": [
    {"area": "...", "concern": "...", "mitigation": "..."}
  ],
  "documents_to_prepare": ["<какие документы потребуются>"]
}
```
